71:       graph = GraphModule::Graph.new
72:       for id in (0..matrix.length-1)
73:         graph.add_node(GraphModule::Node.new(id))
74:       end
75:       matrix.each_index { |node_A_index|
76:         matrix[node_A_index].each_index { |node_B_index|
77:           weight = matrix[node_A_index][node_B_index]
78:           if (weight != 0)          
79:             target_node = graph.get_node(node_A_index)
80:             source_node = graph.get_node(node_B_index)

82:             graph.add_edge(edge)
83:           end
84:         }
85:       }
