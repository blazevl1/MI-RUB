56:     @actual_node = @state_space.root
57:     while (@matrix.rows != 0 || @matrix.columns != 0)
58:       progress = false
59:       @matrix.reset_row_header_iterator
60:       while(@matrix.row_header_has_next?)
61:         value = @matrix.row_header_next
62:         if (not @actual_node.has_child_with_value?(value))
63:           node = Node.new(value)
64:           @actual_node.add_child(node)
65:           @actual_node = node
66:           @matrix.disconnect_section(@actual_node.value)
67:           progress = true
68:           break
69:         end
70:       end
71:       if (not progress)
72:         can_backtrack = backtrack
